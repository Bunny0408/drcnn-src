# gui code 1

import tkinter as tk
from tkinter import filedialog
from PIL import Image, ImageTk

# Create a Tkinter window
window = tk.Tk()
window.geometry("400x400")
window.title("DRCNN Image Denoising")

# Create a function to open an image file
def open_image():
    # Get the file path from the user
    file_path = filedialog.askopenfilename(title="Open Image File",
                                           filetypes=(("JPEG files", "*.jpg"),
                                                      ("PNG files", "*.png"),
                                                      ("All files", "*.*")))
    # Load the image file using Pillow
    image = Image.open(file_path)
    # Convert the image to Tkinter-compatible format
    image_tk = ImageTk.PhotoImage(image)
    # Display the image in the GUI
    label_image.config(image=image_tk)
    label_image.image = image_tk

# Create a button to open an image file
button_open_image = tk.Button(window, text="Open Image", command=open_image)
button_open_image.pack()

# Create a label to display the loaded image
label_image = tk.Label(window)
label_image.pack()

# Run the Tkinter event loop
window.mainloop()

import tkinter as tk
from tkinter import filedialog
from PIL import Image, ImageTk

# Create a Tkinter window
window = tk.Tk()
window.geometry("400x500")
window.title("DRCNN Image Denoising")

# Create a function to open an image file
def open_image():
    # Get the file path from the user
    file_path = filedialog.askopenfilename(title="Open Image File",
                                           filetypes=(("JPEG files", "*.jpg"),
                                                      ("PNG files", "*.png"),
                                                      ("All files", "*.*")))
    # Load the image file using Pillow
    image = Image.open(file_path)
    # Convert the image to Tkinter-compatible format
    image_tk = ImageTk.PhotoImage(image)
    # Display the image in the GUI
    label_image.config(image=image_tk)
    label_image.image = image_tk

# Create a button to open an image file
button_open_image = tk.Button(window, text="Open Image", command=open_image)
button_open_image.pack()

# Create a label to display the loaded image
label_image = tk.Label(window)
label_image.pack()

# Create three text boxes for user input
label_input = tk.Label(window, text="Input values:")
label_input.pack()

label_textbox1 = tk.Label(window, text="Textbox 1:")
label_textbox1.pack()
textbox1 = tk.Entry(window)
textbox1.pack()

label_textbox2 = tk.Label(window, text="Textbox 2:")
label_textbox2.pack()
textbox2 = tk.Entry(window)
textbox2.pack()

label_textbox3 = tk.Label(window, text="Textbox 3:")
label_textbox3.pack()
textbox3 = tk.Entry(window)
textbox3.pack()

# Create a radio button for user input
label_radio = tk.Label(window, text="Select an option:")
label_radio.pack()

radio_var = tk.StringVar()
radio_var.set("Option 1")

radio_button1 = tk.Radiobutton(window, text="Option 1", variable=radio_var, value="Option 1")
radio_button1.pack()

radio_button2 = tk.Radiobutton(window, text="Option 2", variable=radio_var, value="Option 2")
radio_button2.pack()

# Create a submit button to submit user inputs
def submit():
    textbox1_value = textbox1.get()
    textbox2_value = textbox2.get()
    textbox3_value = textbox3.get()
    radio_value = radio_var.get()
    print(f"Textbox 1: {textbox1_value}\nTextbox 2: {textbox2_value}\nTextbox 3: {textbox3_value}\nRadio button value: {radio_value}")

button_submit = tk.Button(window, text="Submit", command=submit)
button_submit.pack()

# Run the Tkinter event loop
window.mainloop()
